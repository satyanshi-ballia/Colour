🧠 3. script.js

javascript
const basket = document.getElementById("basket");
const star = document.getElementById("star");
const scoreDisplay = document.getElementById("score");
let score = 0;

let basketPos = 170;
document.addEventListener("keydown", (e) => {
  if (e.key === "ArrowLeft" && basketPos > 0) {
    basketPos -= 20;
  } else if (e.key === "ArrowRight" && basketPos < 340) {
    basketPos += 20;
  }
  basket.style.left = basketPos + "px";
});

// Reset star after it reaches the bottom
function checkCatch() {
  const starTop = parseInt(window.getComputedStyle(star).top);
  const starLeft = parseInt(window.getComputedStyle(star).left);
  const basketLeft = parseInt(basket.style.left);

  if (starTop >= 470 && Math.abs(starLeft - basketLeft) < 40) {
    score++;
    scoreDisplay.textContent = score;
    resetStar();
  } else if (starTop >= 500) {
    resetStar(); // missed it
  }
}

function resetStar() {
  star.style.top = "-30px";
  star.style.left = Math.floor(Math.random() * 370) + "px";
}

function moveStar() {
  let top = parseInt(window.getComputedStyle(star).top);
  star.style.top = top + 5 + "px";
  checkCatch();
  requestAnimationFrame(moveStar);
}

resetStar();
moveStar();
